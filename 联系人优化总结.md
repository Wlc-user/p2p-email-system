## 技术细节

### 节点ID验证逻辑
```javascript
// 1. 长度检查
if (nodeId.length !== 40) {
  return '节点ID必须是40位字符';
}

// 2. 字符类型检查
if (!/^[0-9a-fA-F]+$/.test(nodeId)) {
  return '节点ID只能包含0-9和a-f（或A-F）';
}

// 3. 格式验证
try {
  int(nodeId, 16); // 十六进制转换
} catch (e) {
  return '节点ID格式错误';
}
```

### 剪贴板API
```javascript
// 读取剪贴板
const text = await navigator.clipboard.readText();
const cleaned = text.replace(/[\s\n\r]/g, ''); // 清理

// 写入剪贴板
await navigator.clipboard.writeText(nodeId);
```

### 公网地址检测
```javascript
useEffect(() => {
  const checkNodeInfo = async () => {
    const response = await fetch('/api/node');
    const data = await response.json();

    // 查找公网候选者
    const publicCandidate = data.ice_candidates?.find(
      c => c.type === 'public' || c.type === 'TLS443'
    );

    if (publicCandidate) {
      setPublicAddress(`${host}:${port}`);
    }
  };

  setInterval(checkNodeInfo, 3000);
}, []);
```
