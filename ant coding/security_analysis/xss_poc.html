# security.py 中的防御代码
def sanitize_html_content(html_content: str) -> str:
    """清理HTML内容，移除危险元素和属性"""
    
    from bs4 import BeautifulSoup
    import bleach
    
    # 允许的安全标签
    ALLOWED_TAGS = [
        'p', 'br', 'b', 'i', 'u', 'strong', 'em', 'blockquote',
        'code', 'pre', 'h1', 'h2', 'h3', 'h4', 'h5', 'h6',
        'ul', 'ol', 'li', 'a', 'img', 'span', 'div'
    ]
    
    # 允许的安全属性
    ALLOWED_ATTRIBUTES = {
        'a': ['href', 'title', 'target'],
        'img': ['src', 'alt', 'title', 'width', 'height'],
        '*': ['class', 'style']  # 需要额外清理style属性
    }
    
    # 清理style属性中的危险内容
    def clean_style(style_value):
        # 移除expression、javascript、url等危险内容
        dangerous_patterns = [
            r'expression\(.*?\)',
            r'javascript:',
            r'url\(.*?\)',
            r'@import',
            r'behavior:'
        ]
        
        for pattern in dangerous_patterns:
            style_value = re.sub(pattern, '', style_value, flags=re.IGNORECASE)
        
        return style_value
    
    # 使用bleach进行基本清理
    cleaned = bleach.clean(
        html_content,
        tags=ALLOWED_TAGS,
        attributes=ALLOWED_ATTRIBUTES,
        styles=['color', 'background-color', 'font-size', 'text-align'],
        strip=True
    )
    
    # 使用BeautifulSoup进行进一步清理
    soup = BeautifulSoup(cleaned, 'html.parser')
    
    # 移除所有事件处理器属性
    event_handlers = [
        'onclick', 'onmouseover', 'onmouseout', 'onload', 'onerror',
        'onsubmit', 'onchange', 'onfocus', 'onblur', 'onkeydown',
        'onkeypress', 'onkeyup'
    ]
    
    for tag in soup.find_all(True):
        # 移除事件处理器
        for attr in list(tag.attrs):
            if attr.lower().startswith('on'):
                del tag[attr]
            
            # 清理style属性
            if attr == 'style' and tag[attr]:
                tag[attr] = clean_style(tag[attr])
            
            # 清理href和src属性
            if attr in ['href', 'src']:
                value = tag[attr]
                if isinstance(value, str) and value.lower().startswith(('javascript:', 'data:', 'vbscript:')):
                    del tag[attr]
    
    # 移除script、iframe、object、embed等危险标签
    for dangerous in ['script', 'iframe', 'object', 'embed', 'frame', 'frameset', 'base']:
        for tag in soup.find_all(dangerous):
            tag.decompose()
    
    return str(soup)

📊 攻击影响评估

攻击类型 影响范围 严重程度 防御措施
会话劫持 单个用户账户 🔴 严重 HttpOnly Cookie、CSP、输入清理
数据窃取 用户邮件数据 🔴 严重 内容过滤、输出编码
账户篡改 用户账户设置 🔴 严重 CSRF令牌、操作确认
钓鱼攻击 多个用户 🟡 中等 链接验证、用户教育
键盘记录 用户输入信息 🔴 严重 事件处理器过滤

🚀 完整防御方案

  1. 输入验证：对所有用户输入进行严格验证
  2. 输出编码：在显示前对特殊字符进行编码
  3. 内容清理：使用HTML净化库清理邮件内容
  4. CSP头：实施严格的内容安全策略
  5. HttpOnly Cookie：防止JavaScript访问敏感Cookie
  6. 沙箱隔离：将用户内容放在沙箱iframe中显示
  7. 附件扫描：扫描所有附件中的恶意内容
  8. 用户教育：提醒用户注意可疑邮件
